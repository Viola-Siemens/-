121  private static void SetPositionVisibleinSceneView(RectTransform canvasRTransform, RectTransform itemTransform)
122  {
123  // Find the best scene view
124  SceneView sceneView = SceneView.lastActiveSceneView;
125  if (sceneView == null && SceneView.sceneViews.Count > 0)
126  sceneView = SceneView.sceneViews[0] as SceneView;
127 
128  // Couldn't find a SceneView. Don't set position.
129  if (sceneView == null || sceneView.camera == null)
130  return;
131 
132  // Create world space Plane from canvas position.
133  Camera camera = sceneView.camera;
134  Vector3 position = Vector3.zero;
135  if (RectTransformUtility.ScreenPointToLocalPointInRectangle(canvasRTransform, new Vector2(camera.pixelWidth / 2, camera.pixelHeight / 2), camera, out Vector2 localPlanePosition))
136  {
137  // Adjust for canvas pivot
138  localPlanePosition.x = localPlanePosition.x + canvasRTransform.sizeDelta.x * canvasRTransform.pivot.x;
139  localPlanePosition.y = localPlanePosition.y + canvasRTransform.sizeDelta.y * canvasRTransform.pivot.y;
140 
141  localPlanePosition.x = Mathf.Clamp(localPlanePosition.x, 0, canvasRTransform.sizeDelta.x);
142  localPlanePosition.y = Mathf.Clamp(localPlanePosition.y, 0, canvasRTransform.sizeDelta.y);
143 
144  // Adjust for anchoring
145  position.x = localPlanePosition.x - canvasRTransform.sizeDelta.x * itemTransform.anchorMin.x;
146  position.y = localPlanePosition.y - canvasRTransform.sizeDelta.y * itemTransform.anchorMin.y;
147 
148  Vector3 minLocalPosition;
149  minLocalPosition.x = canvasRTransform.sizeDelta.x * (0 - canvasRTransform.pivot.x) + itemTransform.sizeDelta.x * itemTransform.pivot.x;
150  minLocalPosition.y = canvasRTransform.sizeDelta.y * (0 - canvasRTransform.pivot.y) + itemTransform.sizeDelta.y * itemTransform.pivot.y;
151 
152  Vector3 maxLocalPosition;
153  maxLocalPosition.x = canvasRTransform.sizeDelta.x * (1 - canvasRTransform.pivot.x) - itemTransform.sizeDelta.x * itemTransform.pivot.x;
154  maxLocalPosition.y = canvasRTransform.sizeDelta.y * (1 - canvasRTransform.pivot.y) - itemTransform.sizeDelta.y * itemTransform.pivot.y;
155 
156  position.x = Mathf.Clamp(position.x, minLocalPosition.x, maxLocalPosition.x);
157  position.y = Mathf.Clamp(position.y, minLocalPosition.y, maxLocalPosition.y);
158  }
159 
160  itemTransform.anchoredPosition = position;
161  itemTransform.localRotation = Quaternion.identity;
162  itemTransform.localScale = Vector3.one;
163  }

166  private static void PlaceUIElementRoot(GameObject element, MenuCommand menuCommand)
167  {
168  GameObject parent = menuCommand.context as GameObject;
169  bool explicitParentChoice = true;
170  if (parent == null)
171  {
172  parent = GetOrCreateCanvasGameObject();
173  explicitParentChoice = false;
174 
175  // If in Prefab Mode, Canvas has to be part of Prefab contents,
176  // otherwise use Prefab root instead.
177  PrefabStage prefabStage = PrefabStageUtility.GetCurrentPrefabStage();
178  if (prefabStage != null && !prefabStage.IsPartOfPrefabContents(parent))
179  parent = prefabStage.prefabContentsRoot;
180  }
181  if (parent.GetComponentInParent<Canvas>() == null)
182  {
183  // Create canvas under context GameObject,
184  // and make that be the parent which UI element is added under.
185  GameObject canvas = CreateNewUI();
186  canvas.transform.SetParent(parent.transform, false);
187  parent = canvas;
188  }
189 
190  // Setting the element to be a child of an element already in the scene should
191  // be sufficient to also move the element to that scene.
192  // However, it seems the element needs to be already in its destination scene when the
193  // RegisterCreatedObjectUndo is performed; otherwise the scene it was created in is dirtied.
194  SceneManager.MoveGameObjectToScene(element, parent.scene);
195 
196  if (element.transform.parent == null)
197  {
198  Undo.SetTransformParent(element.transform, parent.transform, "Parent " + element.name);
199  }
200 
201  GameObjectUtility.EnsureUniqueNameForSibling(element);
202 
203  // We have to fix up the undo name since the name of the object was only known after reparenting it.
204  Undo.SetCurrentGroupName("Create " + element.name);
205 
206  GameObjectUtility.SetParentAndAlign(element, parent);
207  if (!explicitParentChoice) // not a context click, so center in sceneview
208  SetPositionVisibleinSceneView(parent.GetComponent<RectTransform>(), element.GetComponent<RectTransform>());
209 
210  Undo.RegisterCreatedObjectUndo(element, "Create " + element.name);
211 
212  Selection.activeGameObject = element;
213  }

233  customScene = true;

236  Undo.RegisterCreatedObjectUndo(root, "Create " + root.name);

274  // Helper function that returns a Canvas GameObject; preferably a parent of the selection, or other existing Canvas.
275  static public GameObject GetOrCreateCanvasGameObject()
276  {
277  GameObject selectedGo = Selection.activeGameObject;
278 
279  // Try to find a gameobject that is the selected GO or one if its parents.
280  Canvas canvas = (selectedGo != null) ? selectedGo.GetComponentInParent<Canvas>() : null;
281  if (IsValidCanvas(canvas))
282  return canvas.gameObject;
283 
284  // No canvas in selection or its parents? Then use any valid canvas.
285  // We have to find all loaded Canvases, not just the ones in main scenes.
286  Canvas[] canvasArray = StageUtility.GetCurrentStageHandle().FindComponentsOfType<Canvas>();
287  for (int i = 0; i < canvasArray.Length; i++)
288  if (IsValidCanvas(canvasArray[i]))
289  return canvasArray[i].gameObject;
290 
291  // No canvas in the scene at all? Then create a new one.
292  return CreateNewUI();
293  }

295  static bool IsValidCanvas(Canvas canvas)
296  {
297  if (canvas == null || !canvas.gameObject.activeInHierarchy)
298  return false;
299 
300  // It's important that the non-editable canvas from a prefab scene won't be rejected,
301  // but canvases not visible in the Hierarchy at all do. Don't check for HideAndDontSave.
302  if (EditorUtility.IsPersistent(canvas) || (canvas.hideFlags & HideFlags.HideInHierarchy) != 0)
303  return false;
304 
305  if (StageUtility.GetStageHandle(canvas.gameObject) != StageUtility.GetCurrentStageHandle())
306  return false;
307 
308  return true;
309  }
310  }
