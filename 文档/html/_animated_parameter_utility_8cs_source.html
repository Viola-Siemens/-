108  public static IEnumerable<SerializedProperty> GetAllAnimatableParameters(UnityObject asset)
109  {
110  SerializedObject serializedObject;
111  if (!TryGetSerializedPlayableAsset(asset, out serializedObject))
112  yield break;
113 
114  var prop = serializedObject.GetIterator();
115 
116  // We need to keep this variable because prop starts invalid
117  var outOfBounds = false;
118  while (!outOfBounds && prop.NextVisible(true))
119  {
120  foreach (var property in SelectAnimatableProperty(prop))
121  yield return property;
122 
123  // We can become out of bounds by calling SelectAnimatableProperty, if the last iterated property is a color.
124  outOfBounds = !prop.isValid;
125  }
126  }

128  static IEnumerable<SerializedProperty> SelectAnimatableProperty(SerializedProperty prop)
129  {
130  // We're only interested by animatable leaf parameters
131  if (!prop.hasChildren && IsParameterAnimatable(prop))
132  yield return prop.Copy();
133 
134  // Color type is not considered "visible" when iterating
135  if (prop.propertyType == SerializedPropertyType.Color)
136  {
137  var end = prop.GetEndProperty();
138 
139  // For some reasons, if the last 2+ serialized properties are of type Color, prop becomes invalid and
140  // Next() throws an exception. This is not the case when only the last serialized property is a Color.
141  while (!SerializedProperty.EqualContents(prop, end) && prop.isValid && prop.Next(true))
142  {
143  foreach (var property in SelectAnimatableProperty(prop))
144  yield return property;
145  }
146  }
147  }

196  static bool IsKeyableInHierarchy(SerializedProperty property)
197  {
198  const BindingFlags bindingFlags = BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance;
199  var pathSegments = property.propertyPath.Split('.');
200  var type = property.serializedObject.targetObject.GetType();
201  foreach (var segment in pathSegments)
202  {
203  if (type.GetCustomAttributes(typeof(NotKeyableAttribute), false).Any())
204  {
205  return false;
206  }
207 
208  if (type.IsArray)
209  {
210  if (segment != "Array")
211  type = type.GetElementType();
212  continue;
213  }
214 
215  var fieldInfo = type.GetField(segment, bindingFlags);
216 
217  if (fieldInfo == null ||
218  fieldInfo.GetCustomAttributes(typeof(NotKeyableAttribute), false).Any() ||
219  fieldInfo.GetCustomAttributes(typeof(HideInInspector), false).Any())
220  {
221  return false;
222  }
223 
224  type = fieldInfo.FieldType;
225  }
226 
227  return true;
