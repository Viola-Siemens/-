343  private static UnityEngine.Object GetUnderlyingAsset(WindowState state, TimelineClip clip)
344  {
345  var asset = clip.asset as ScriptableObject;
346  if (asset == null)
347  return null;
348 
349  var fields = ObjectReferenceField.FindObjectReferences(asset.GetType());
350  if (fields.Length == 0)
351  return GetExternalPlayableAsset(clip);
352 
353  // Find the first non-null field
354  foreach (var field in fields)
355  {

358  continue;
359  var obj = field.Find(asset, state.editSequence.director);
