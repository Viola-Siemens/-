253  Event evt = Event.current;

262  break;

265  return true;
266 
267  if ((source[0] == '\\' && (c == 'u' || c == 'U')) == false)
268  evt.character = '\0';
269 
270  break;
271  case 2:
272  case 3:
273  case 4:
274  case 5:
275  if ((c < '0' || c > '9') && (c < 'a' || c > 'f') && (c < 'A' || c > 'F'))
276  evt.character = '\0';
277  break;
278  case 6:
279  case 7:
280  case 8:
281  case 9:
282  if (source[1] == 'u' || (c < '0' || c > '9') && (c < 'a' || c > 'f') && (c < 'A' || c > 'F'))
283  evt.character = '\0';
284 
285  // Validate input
286  if (length == 6 && source[1] == 'u' && source != m_PreviousInput)
287  return true;
288  break;
289  case 10:
290  if (source != m_PreviousInput)
291  return true;
292 
293  evt.character = '\0';
294  break;

298  m_PreviousInput = source;

317  void DrawGlyph(uint glyphIndex, Rect position, SerializedProperty property)
318  {
319  // Get a reference to the sprite texture
320  TMP_FontAsset fontAsset = property.serializedObject.targetObject as TMP_FontAsset;
321 
322  if (fontAsset == null)
323  return;
324 
325  // Check if glyph currently exists in the atlas texture.
326  if (!fontAsset.glyphLookupTable.TryGetValue(glyphIndex, out Glyph glyph))
327  return;
328 
329  // Get reference to atlas texture.
330  int atlasIndex = fontAsset.m_AtlasTextureIndex;
331  Texture2D atlasTexture = fontAsset.atlasTextures.Length > atlasIndex ? fontAsset.atlasTextures[atlasIndex] : null;
332 
333  if (atlasTexture == null)
334  return;
335 
336  Material mat;
337  if (((GlyphRasterModes)fontAsset.atlasRenderMode & GlyphRasterModes.RASTER_MODE_BITMAP) == GlyphRasterModes.RASTER_MODE_BITMAP)
338  {
339  mat = TMP_FontAssetEditor.internalBitmapMaterial;
340 
341  if (mat == null)
342  return;
343 
344  mat.mainTexture = atlasTexture;
345  }
346  else
347  {
348  mat = TMP_FontAssetEditor.internalSDFMaterial;
349 
350  if (mat == null)
351  return;
352 
353  mat.mainTexture = atlasTexture;
354  mat.SetFloat(ShaderUtilities.ID_GradientScale, fontAsset.atlasPadding + 1);
355  }
356 
357  // Draw glyph from atlas texture.
358  Rect glyphDrawPosition = new Rect(position.x, position.y + 2, 64, 60);
359 
360  GlyphRect glyphRect = glyph.glyphRect;
361 
362  float normalizedHeight = fontAsset.faceInfo.ascentLine - fontAsset.faceInfo.descentLine;
363  float scale = glyphDrawPosition.width / normalizedHeight;
364 
365  // Compute the normalized texture coordinates
366  Rect texCoords = new Rect((float)glyphRect.x / atlasTexture.width, (float)glyphRect.y / atlasTexture.height, (float)glyphRect.width / atlasTexture.width, (float)glyphRect.height / atlasTexture.height);
367 
368  if (Event.current.type == EventType.Repaint)
369  {
370  glyphDrawPosition.x += (glyphDrawPosition.width - glyphRect.width * scale) / 2;
371  glyphDrawPosition.y += (glyphDrawPosition.height - glyphRect.height * scale) / 2;
372  glyphDrawPosition.width = glyphRect.width * scale;
373  glyphDrawPosition.height = glyphRect.height * scale;
374 
375  // Could switch to using the default material of the font asset which would require passing scale to the shader.
376  Graphics.DrawTexture(glyphDrawPosition, atlasTexture, texCoords, 0, 0, 0, 0, new Color(1f, 1f, 1f), mat);
