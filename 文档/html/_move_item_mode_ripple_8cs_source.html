139  public void UpdateMove(IEnumerable<ItemsPerTrack> itemsGroups)
140  {
141  if (m_Detached)
142  return;
143 
144  m_TrackLocked = true;
145 
146  var overlap = 0.0;
147  foreach (var itemsGroup in itemsGroups)
148  {
149  var track = itemsGroup.targetTrack;
150  if (track == null) continue;
151 
152  var prevItemInfo = m_PreviousItem[track];
153  if (prevItemInfo.item != null)
154  {
155  var prevItem = prevItemInfo.item;
156  var firstItem = prevItemInfo.firstSelectedItem;
157 
158  if (prevItemInfo.blending)
159  prevItemInfo.blending = prevItem.end > firstItem.start;
160 
161  if (prevItemInfo.blending)
162  {
163  var b = EditModeUtils.BlendDuration(firstItem, TrimEdge.End);
164  overlap = Math.Max(overlap, Math.Max(prevItem.start, prevItem.end - firstItem.end + firstItem.start + b) - firstItem.start);
165  }
166  else
167  {
168  overlap = Math.Max(overlap, prevItem.end - firstItem.start);
169  }
170  }
171  }
172 
173  if (overlap > 0)
174  {
175  foreach (var itemsGroup in itemsGroups)
176  {
177  foreach (var item in itemsGroup.items)
178  item.start += overlap;
179  }
180  }
181 
182  var newEnd = itemsGroups.Max(m => m.items.Max(c => c.end));
183 
184  var offset = newEnd - m_PreviousEnd;
185  m_PreviousEnd = newEnd;
186 
187  foreach (var itemsGroup in itemsGroups)
188  {
189  foreach (var item in m_NextItems[itemsGroup.targetTrack])
190  item.start += offset;
191  }

194  static PrevItemInfo CalculatePrevItemInfo(List<ITimelineItem> orderedSelection, List<ITimelineItem> orderedSiblings, IEnumerable<Type> itemTypes)
195  {
196  ITimelineItem previousItem = null;
197  ITimelineItem firstSelectedItem = null;
198  var gap = double.PositiveInfinity;
199 
200  foreach (var type in itemTypes)
201  {
202  var firstSelectedItemOfType = orderedSelection.FirstOrDefault(i => i.GetType() == type);
203  if (firstSelectedItemOfType == null) continue;
204 
205  var previousItemOfType = orderedSiblings.LastOrDefault(i => i.GetType() == type && i.start < firstSelectedItemOfType.start);
206  if (previousItemOfType == null) continue;
207 
208  var currentGap = firstSelectedItemOfType.start - previousItemOfType.end;
209  if (currentGap < gap)
210  {
211  gap = currentGap;
212  firstSelectedItem = firstSelectedItemOfType;
213  previousItem = previousItemOfType;
214  }
215  }
216 
217  return new PrevItemInfo(previousItem, firstSelectedItem);
218  }
