108  static string GetPackageFullPath()
109  {
110  // Check for potential UPM package
111  string packagePath = Path.GetFullPath("Packages/com.unity.textmeshpro");
112  if (Directory.Exists(packagePath))
113  {
114  return packagePath;
115  }
116 
117  packagePath = Path.GetFullPath("Assets/..");
118  if (Directory.Exists(packagePath))
119  {
120  // Search default location for development package
121  if (Directory.Exists(packagePath + "/Assets/Packages/com.unity.TextMeshPro/Editor Resources"))
122  {
123  return packagePath + "/Assets/Packages/com.unity.TextMeshPro";
124  }
125 
126  // Search for default location of normal TextMesh Pro AssetStore package
127  if (Directory.Exists(packagePath + "/Assets/TextMesh Pro/Editor Resources"))
128  {
129  return packagePath + "/Assets/TextMesh Pro";
130  }
131 
132  // Search for potential alternative locations in the user project
133  string[] matchingPaths = Directory.GetDirectories(packagePath, "TextMesh Pro", SearchOption.AllDirectories);
134  string path = ValidateLocation(matchingPaths, packagePath);
135  if (path != null) return packagePath + path;
136  }
137 
138  return null;
139  }

141  static string ValidateLocation(string[] paths, string projectPath)
142  {

145  // Check if the Editor Resources folder exists.
146  if (Directory.Exists(paths[i] + "/Editor Resources"))
