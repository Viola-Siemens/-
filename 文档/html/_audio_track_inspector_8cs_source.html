77  void DrawMixInfoSection()
78  {
79  if (m_Director == null || targets.Length > 1)
80  return;
81 
82  var binding = m_Director.GetGenericBinding(target) as AudioSource;
83  if (binding == null)
84  return;
85 
86  var audioSourceVolume = binding.volume;
87  var audioSourcePan = binding.panStereo;
88  var audioSourceBlend = binding.spatialBlend;
89 
90  var trackVolume = m_VolumeProperty.floatValue;
91  var trackPan = m_StereoPanProperty.floatValue;
92  var trackBlend = m_SpatialBlendProperty.floatValue;
93 
94  // Skip sections when result is obvious
95 
96  var skipVolumeInfo = Math.Abs(audioSourceVolume) < float.Epsilon && Math.Abs(trackVolume) < float.Epsilon || // All muted
97  Math.Abs(audioSourceVolume - 1) < float.Epsilon && Math.Abs(trackVolume - 1) < float.Epsilon; // All max volume
98 
99  var skipPanInfo = Math.Abs(audioSourcePan) < float.Epsilon && Math.Abs(trackPan) < float.Epsilon || // All centered
100  Math.Abs(audioSourcePan - 1) < float.Epsilon && Math.Abs(trackPan - 1) < float.Epsilon || // All right
101  Math.Abs(audioSourcePan - (-1.0f)) < float.Epsilon && Math.Abs(trackPan - (-1.0f)) < float.Epsilon; // All left
102 
103  var skipBlendInfo = Math.Abs(audioSourceBlend) < float.Epsilon && Math.Abs(trackBlend) < float.Epsilon || // All 2D
104  Math.Abs(audioSourceBlend - 1) < float.Epsilon && Math.Abs(trackBlend - 1) < float.Epsilon; // All 3D
105 
106  if (skipVolumeInfo && skipPanInfo && skipBlendInfo)
107  return;
108 
109  s_MixInfoBuilder.Length = 0;
110 
111  if (!skipVolumeInfo)
112  s_MixInfoBuilder.AppendFormat(
113  Styles.mixedPropertiesInfo,
114  (audioSourceVolume * trackVolume).ToString(Styles.valuesFormatter, CultureInfo.InvariantCulture),
115  trackVolume.ToString(Styles.valuesFormatter, CultureInfo.InvariantCulture),
116  audioSourceVolume.ToString(Styles.valuesFormatter, CultureInfo.InvariantCulture),
117  AudioSourceInspector.Styles.volumeLabel.text);
118 
119  if (!skipVolumeInfo && !skipPanInfo)
120  s_MixInfoBuilder.Append(Styles.mixInfoSectionSeparator);
121 
122  if (!skipPanInfo)
123  s_MixInfoBuilder.AppendFormat(
124  Styles.mixedPropertiesInfo,
125  Mathf.Clamp(audioSourcePan + trackPan, -1.0f, 1.0f).ToString(Styles.valuesFormatter, CultureInfo.InvariantCulture),
126  trackPan.ToString(Styles.valuesFormatter, CultureInfo.InvariantCulture),
127  audioSourcePan.ToString(Styles.valuesFormatter, CultureInfo.InvariantCulture),
128  AudioSourceInspector.Styles.panStereoLabel.text);
129 
130  if ((!skipVolumeInfo || !skipPanInfo) && !skipBlendInfo)
131  s_MixInfoBuilder.Append(Styles.mixInfoSectionSeparator);
132 
133  if (!skipBlendInfo)
134  s_MixInfoBuilder.AppendFormat(
135  Styles.mixedPropertiesInfo,
136  Mathf.Clamp01(audioSourceBlend + trackBlend).ToString(Styles.valuesFormatter, CultureInfo.InvariantCulture),
137  trackBlend.ToString(Styles.valuesFormatter, CultureInfo.InvariantCulture),
138  audioSourceBlend.ToString(Styles.valuesFormatter, CultureInfo.InvariantCulture),
139  AudioSourceInspector.Styles.spatialBlendLabel.text);
140 
141  EditorGUILayout.Space();
142  EditorGUILayout.HelpBox(new GUIContent(s_MixInfoBuilder.ToString()));
143  }
