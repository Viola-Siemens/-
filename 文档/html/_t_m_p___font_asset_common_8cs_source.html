402  private static TMP_FontAsset SearchForCharacterInternal(TMP_FontAsset font, uint unicode, out TMP_Character character)
403  {
404  character = null;
405 
406  if (font == null) return null;
407 
408  if (font.characterLookupTable.TryGetValue(unicode, out character))
409  {
410  return font;
411  }
412  else if (font.fallbackFontAssetTable != null && font.fallbackFontAssetTable.Count > 0)
413  {
414  for (int i = 0; i < font.fallbackFontAssetTable.Count && character == null; i++)
415  {
416  TMP_FontAsset temp = font.fallbackFontAssetTable[i];
417  if (temp == null) continue;
418 
419  int id = temp.GetInstanceID();
420 
421  // Skip over the fallback font asset in the event it is null or if already searched.
422  if (k_searchedFontAssets.Contains(id)) continue;
423 
424  // Add to list of font assets already searched.
425  k_searchedFontAssets.Add(id);
426 
427  temp = SearchForCharacterInternal(temp, unicode, out character);
428 
429  if (temp != null)
430  return temp;
431  }
432  }
433 
434  return null;
435  }
