12  public static void Insert(TimelineAsset asset, double at, double amount, double tolerance)
13  {
14  // gather all clips
15  var clips = asset.flattenedTracks.SelectMany(x => x.clips).Where(x => (x.start - at) >= -tolerance).ToList();
16  var markers = asset.flattenedTracks.SelectMany(x => x.GetMarkers()).Where(x => (x.time - at) >= -tolerance).ToList();
17 
18  if (!clips.Any() && !markers.Any())
19  return;
20 
21  // push undo on the tracks for the clips that are being modified
22  foreach (var t in clips.Select(x => x.parentTrack).Distinct())
23  {
24  TimelineUndo.PushUndo(t, kInsertTime);
25  }
26 
27  // push the clips
28  foreach (var clip in clips)
29  {
30  clip.start += amount;
31  }
32 
33  // push undos and move the markers
34  foreach (var marker in markers)
35  {
36  var obj = marker as UnityEngine.Object;
37  if (obj != null)
38  TimelineUndo.PushUndo(obj, kInsertTime);
39  marker.time += amount;
40  }
41 
42  TimelineEditor.Refresh(RefreshReason.ContentsModified);
43  }
44  }
