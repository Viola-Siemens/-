557  string ProjectText(Assembly assembly,
558  Dictionary<string, string> allAssetsProjectParts,
559  List<ResponseFileData> responseFilesData,
560  List<Assembly> allProjectIslands)
561  {
562  var projectBuilder = new StringBuilder(ProjectHeader(assembly, responseFilesData));
563  var references = new List<string>();
564  var projectReferences = new List<Match>();
565 
566  foreach (string file in assembly.sourceFiles)
567  {
568  if (!ShouldFileBePartOfSolution(file))
569  continue;
570 
571  var extension = Path.GetExtension(file).ToLower();
572  var fullFile = EscapedRelativePathFor(file);
573  if (".dll" != extension)
574  {
575  projectBuilder.Append(" <Compile Include=\"").Append(fullFile).Append("\" />").Append(Environment.NewLine);
576  }
577  else
578  {
579  references.Add(fullFile);
580  }
581  }
582 
583  // Append additional non-script files that should be included in project generation.
584  if (allAssetsProjectParts.TryGetValue(assembly.name, out var additionalAssetsForProject))
585  projectBuilder.Append(additionalAssetsForProject);
586 
587  var islandRefs = references.Union(assembly.allReferences);
588  foreach (string reference in islandRefs)
589  {
590  if (reference.EndsWith("/UnityEditor.dll", StringComparison.Ordinal)
591  || reference.EndsWith("/UnityEngine.dll", StringComparison.Ordinal)
592  || reference.EndsWith("\\UnityEditor.dll", StringComparison.Ordinal)
593  || reference.EndsWith("\\UnityEngine.dll", StringComparison.Ordinal))
594  continue;
595 
596  var match = k_ScriptReferenceExpression.Match(reference);
597  if (match.Success)
598  {
599  // assume csharp language
600  // Add a reference to a project except if it's a reference to a script assembly
601  // that we are not generating a project for. This will be the case for assemblies
602  // coming from .assembly.json files in non-internalized packages.
603  var dllName = match.Groups["dllname"].Value;
604  if (allProjectIslands.Any(i => Path.GetFileName(i.outputPath) == dllName))
605  {
606  projectReferences.Add(match);
607  continue;
608  }
609  }
610 
611  string fullReference = Path.IsPathRooted(reference) ? reference : Path.Combine(ProjectDirectory, reference);
612 
613  AppendReference(fullReference, projectBuilder);
614  }
615 
616  var responseRefs = responseFilesData.SelectMany(x => x.FullPathReferences.Select(r => r));
617  foreach (var reference in responseRefs)
618  {
619  AppendReference(reference, projectBuilder);
620  }
621 
622  if (0 < projectReferences.Count)
623  {
624  projectBuilder.AppendLine(" </ItemGroup>");
625  projectBuilder.AppendLine(" <ItemGroup>");
626  foreach (Match reference in projectReferences)
627  {
628  var referencedProject = reference.Groups["project"].Value;
629 
630  projectBuilder.Append(" <ProjectReference Include=\"").Append(referencedProject)
631  .Append(GetProjectExtension()).Append("\">").Append(Environment.NewLine);
632  projectBuilder
633  .Append(" <Project>{")
634  .Append(m_GUIDGenerator.ProjectGuid(m_ProjectName, reference.Groups["project"].Value))
635  .Append("}</Project>")
636  .Append(Environment.NewLine);
637  projectBuilder.Append(" <Name>").Append(referencedProject).Append("</Name>").Append(Environment.NewLine);
638  projectBuilder.AppendLine(" </ProjectReference>");
639  }
640  }
641 
642  projectBuilder.Append(ProjectFooter());
643  return projectBuilder.ToString();
644  }
