141  public static void SetStart(TimelineClip clip, double time)
142  {
143  var supportsClipIn = clip.SupportsClipIn();
144  var supportsPadding = TimelineUtility.IsRecordableAnimationClip(clip);
145 
146  // treat empty recordable clips as not supporting clip in (there are no keys to modify)
147  if (supportsPadding && (clip.animationClip == null || clip.animationClip.empty))
148  {
149  supportsClipIn = false;
150  }
151 
152  if (supportsClipIn && !supportsPadding)
153  {
154  var minStart = clip.FromLocalTimeUnbound(0.0);
155  if (time < minStart)
156  time = minStart;
157  }
158 
159  var maxStart = clip.end - TimelineClip.kMinDuration;
160  if (time > maxStart)
161  time = maxStart;
162 
163  var timeOffset = time - clip.start;
164  var duration = clip.duration - timeOffset;
165 
166  if (supportsClipIn)
167  {
168  if (supportsPadding)
169  {
170  double clipInGlobal = clip.clipIn / clip.timeScale;
171  double keyShift = -timeOffset;
172  if (timeOffset < 0) // left drag, eliminate clipIn before shifting
173  {
174  double clipInDelta = Math.Max(-clipInGlobal, timeOffset);
175  keyShift = -Math.Min(0, timeOffset - clipInDelta);
176  clip.clipIn += clipInDelta * clip.timeScale;
177  }
178  else if (timeOffset > 0) // right drag, elimate padding in animation clip before adding clip in
179  {
180  var clipInfo = AnimationClipCurveCache.Instance.GetCurveInfo(clip.animationClip);
181  double keyDelta = clip.FromLocalTimeUnbound(clipInfo.keyTimes.Min()) - clip.start;
182  keyShift = -Math.Max(0, Math.Min(timeOffset, keyDelta));
183  clip.clipIn += Math.Max(timeOffset + keyShift, 0) * clip.timeScale;
184  }
185  if (keyShift != 0)
186  {
187  AnimationTrackRecorder.ShiftAnimationClip(clip.animationClip, (float)(keyShift * clip.timeScale));
188  }
189  }
190  else
191  {
192  clip.clipIn += timeOffset * clip.timeScale;
193  }
194  }
195 
196  clip.start = time;
197  clip.duration = duration;
198  }

316  static void FixLoops(TimelineClip clip, bool completeLastLoop)
317  {
318  if (!TimelineHelpers.HasUsableAssetDuration(clip))
319  return;
320 
321  var loopDuration = TimelineHelpers.GetLoopDuration(clip);
322  var firstLoopDuration = loopDuration - clip.clipIn * (1.0 / clip.timeScale);
323 
324  // Making sure we don't trim to zero
325  if (!completeLastLoop && firstLoopDuration > clip.duration)
326  return;
327 
328  var numLoops = (clip.duration - firstLoopDuration) / loopDuration;
329  var numCompletedLoops = Math.Floor(numLoops);
330 
331  if (!(numCompletedLoops < numLoops))
332  return;
333 
334  if (completeLastLoop)
335  numCompletedLoops += 1;
336 
337  var newEnd = clip.start + firstLoopDuration + loopDuration * numCompletedLoops;
338 
339  TimelineUndo.PushUndo(clip.parentTrack, "Trim Clip Last Loop");
340 
341  TrimClipWithEditMode(clip, TrimEdge.End, newEnd);
342  }
