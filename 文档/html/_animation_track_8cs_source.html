553  internal override Playable OnCreateClipPlayableGraph(PlayableGraph graph, GameObject go, IntervalTree<RuntimeElement> tree)
554  {
555  if (isSubTrack)
556  throw new InvalidOperationException("Nested animation tracks should never be asked to create a graph directly");
557 
558  List<AnimationTrack> flattenTracks = new List<AnimationTrack>();
559  if (CanCompileClips())
560  flattenTracks.Add(this);
561 
562 
563  bool animatesRootTransform = AnimatesRootTransform();
564  foreach (var subTrack in GetChildTracks())
565  {
566  var child = subTrack as AnimationTrack;
567  if (child != null && child.CanCompileClips())
568  {
569  animatesRootTransform |= child.AnimatesRootTransform();
570  flattenTracks.Add(child);
571  }
572  }
573 
574  // figure out which mode to apply
575  AppliedOffsetMode mode = GetOffsetMode(go, animatesRootTransform);
576  var layerMixer = CreateGroupMixer(graph, go, flattenTracks.Count);
577  for (int c = 0; c < flattenTracks.Count; c++)
578  {
579  var compiledTrackPlayable = flattenTracks[c].inClipMode ?
580  CompileTrackPlayable(graph, flattenTracks[c], go, tree, mode) :
581  flattenTracks[c].CreateInfiniteTrackPlayable(graph, go, tree, mode);
582  graph.Connect(compiledTrackPlayable, 0, layerMixer, c);
583  layerMixer.SetInputWeight(c, flattenTracks[c].inClipMode ? 0 : 1);
584  if (flattenTracks[c].applyAvatarMask && flattenTracks[c].avatarMask != null)
585  {
586  layerMixer.SetLayerMaskFromAvatarMask((uint)c, flattenTracks[c].avatarMask);
587  }
588  }
589 
590  bool requiresMotionXPlayable = RequiresMotionXPlayable(mode, go);
591 
592  Playable mixer = layerMixer;
593  mixer = CreateDefaultBlend(graph, go, mixer, requiresMotionXPlayable);
594 
595  // motionX playable not required in scene offset mode, or root transform mode
596  if (requiresMotionXPlayable)
597  {
598  // If we are animating a root transform, add the motionX to delta playable as the root node
599  var motionXToDelta = AnimationMotionXToDeltaPlayable.Create(graph);
600  graph.Connect(mixer, 0, motionXToDelta, 0);
601  motionXToDelta.SetInputWeight(0, 1.0f);
602  motionXToDelta.SetAbsoluteMotion(UsesAbsoluteMotion(mode));
603  mixer = (Playable)motionXToDelta;
604  }
605 
606 
607 #if UNITY_EDITOR
608  if (!Application.isPlaying)
609  {
610  var animator = GetBinding(go != null ? go.GetComponent<PlayableDirector>() : null);
611  if (animator != null)
612  {
613  GameObject targetGO = animator.gameObject;
614  IAnimationWindowPreview[] previewComponents = targetGO.GetComponents<IAnimationWindowPreview>();
615 
616  m_HasPreviewComponents = previewComponents.Length > 0;
617  if (m_HasPreviewComponents)
618  {
619  foreach (var component in previewComponents)
620  {
621  mixer = component.BuildPreviewGraph(graph, mixer);
622  }
623  }
624  }
625  }
626 #endif
627 
628  return mixer;
629  }
