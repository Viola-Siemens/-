33  public static void CloneExposedReferences(ScriptableObject clone, IExposedPropertyTable sourceTable, IExposedPropertyTable destTable)
34  {
35  var cloneObject = new SerializedObject(clone);
36  SerializedProperty prop = cloneObject.GetIterator();
37  while (prop.Next(true))
38  {
39  if (prop.propertyType == SerializedPropertyType.ExposedReference)
40  {
41  var exposedNameProp = prop.FindPropertyRelative("exposedName");
42  var sourceKey = exposedNameProp.stringValue;
43  var destKey = sourceKey;
44 
45  if (!IsExposedReferenceExplicitlyNamed(sourceKey))
46  destKey = GenerateExposedReferenceName();
47 
48  exposedNameProp.stringValue = destKey;
49 
50  var requiresCopy = sourceTable != destTable || sourceKey != destKey;
51  if (requiresCopy && sourceTable != null && destTable != null)
52  {
53  var valid = false;
54  var target = sourceTable.GetReferenceValue(sourceKey, out valid);
55  if (valid && target != null)
56  {
57  var existing = destTable.GetReferenceValue(destKey, out valid);
58  if (!valid || existing != target)
59  {
60  var destTableObj = destTable as UnityEngine.Object;
61  if (destTableObj != null)
62  TimelineUndo.PushUndo(destTableObj, "Create Clip");
63  destTable.SetReferenceValue(destKey, target);
64  }
65  }
66  }
67  }
68  }
69  cloneObject.ApplyModifiedPropertiesWithoutUndo();
70  }

480  // Note that if clip-in is bigger than the loopDuration, negative times will be added
481  public static double[] GetLoopTimes(TimelineClip clip)

770  public static void RangeSelect<T>(IList<T> totalCollection, IList<T> currentSelection, T clickedItem, Action<T> selector, Action<T> remover) where T : class
771  {
772  var firstSelect = currentSelection.FirstOrDefault();
773  if (firstSelect == null)
774  {
775  selector(clickedItem);
776  return;
777  }
778 
779  var idxFirstSelect = totalCollection.IndexOf(firstSelect);
780  var idxLastSelect = totalCollection.IndexOf(currentSelection.Last());
781  var idxClicked = totalCollection.IndexOf(clickedItem);
782 
783  //case 927807: selection is invalid
784  if (idxFirstSelect < 0)
785  {
786  SelectionManager.Clear();
787  selector(clickedItem);
788  return;
789  }
790 
791  // Expand the selection between the first selected clip and clicked clip (insertion order is important)
792  if (idxFirstSelect < idxClicked)
793  for (var i = idxFirstSelect; i <= idxClicked; ++i)
794  selector(totalCollection[i]);
795  else
796  for (var i = idxFirstSelect; i >= idxClicked; --i)
797  selector(totalCollection[i]);
798 
799  // If clicked inside the selected range, shrink the selection between the the click and last selected clip
800  if (Math.Min(idxFirstSelect, idxLastSelect) < idxClicked && idxClicked < Math.Max(idxFirstSelect, idxLastSelect))
801  for (var i = Math.Min(idxLastSelect, idxClicked); i <= Math.Max(idxLastSelect, idxClicked); ++i)
802  remover(totalCollection[i]);
803 
804  // Ensure clicked clip is selected
805  selector(clickedItem);
806  }
