266  private static string[] CollectPathsFromToolbox(string toolboxRiderRootPath, string dirName, string searchPattern,
267  bool isMac)
268  {
269  if (!Directory.Exists(toolboxRiderRootPath))
270  return new string[0];
271 
272  var channelDirs = Directory.GetDirectories(toolboxRiderRootPath);
273  var paths = channelDirs.SelectMany(channelDir =>
274  {
275  try
276  {
277  // use history.json - last entry stands for the active build https://jetbrains.slack.com/archives/C07KNP99D/p1547807024066500?thread_ts=1547731708.057700&cid=C07KNP99D
278  var historyFile = Path.Combine(channelDir, ".history.json");
279  if (File.Exists(historyFile))
280  {
281  var json = File.ReadAllText(historyFile);
282  var build = ToolboxHistory.GetLatestBuildFromJson(json);
283  if (build != null)
284  {
285  var buildDir = Path.Combine(channelDir, build);
286  var executablePaths = GetExecutablePaths(dirName, searchPattern, isMac, buildDir);
287  if (executablePaths.Any())
288  return executablePaths;
289  }
290  }
291 
292  var channelFile = Path.Combine(channelDir, ".channel.settings.json");
293  if (File.Exists(channelFile))
294  {
295  var json = File.ReadAllText(channelFile).Replace("active-application", "active_application");
296  var build = ToolboxInstallData.GetLatestBuildFromJson(json);
297  if (build != null)
298  {
299  var buildDir = Path.Combine(channelDir, build);
300  var executablePaths = GetExecutablePaths(dirName, searchPattern, isMac, buildDir);
301  if (executablePaths.Any())
302  return executablePaths;
303  }
304  }
305 
306  // changes in toolbox json files format may brake the logic above, so return all found Rider installations
307  return Directory.GetDirectories(channelDir)
308  .SelectMany(buildDir => GetExecutablePaths(dirName, searchPattern, isMac, buildDir));
309  }
310  catch (Exception e)
311  {
312  // do not write to Debug.Log, just log it.
313  Logger.Warn($"Failed to get RiderPath from {channelDir}", e);
314  }
315 
316  return new string[0];
317  })
318  .Where(c => !string.IsNullOrEmpty(c))
319  .ToArray();
320  return paths;
321  }
