113  void DrawTrackStateBox(Rect trackRect, TrackAsset track)

115  const float k_LockTextPadding = 40f;
116  var styles = DirectorStyles.Instance;
117 
118  bool locked = track.locked && !TimelineUtility.IsLockedFromGroup(track);
119  bool muted = track.muted && !TimelineUtility.IsParentMuted(track);
120  bool allSubTrackMuted = TimelineUtility.IsAllSubTrackMuted(track);
121 
122  GUIContent content = null;
123  if (locked && muted)
124  {
125  content = Styles.s_LockedAndMuted;
126  if (!allSubTrackMuted)
127  content = Styles.s_LockedAndPartiallyMuted;
128  }
129  else if (locked) content = Styles.s_Locked;
130  else if (muted)
131  {
132  content = Styles.s_Muted;
133  if (!allSubTrackMuted)
134  content = Styles.s_PartiallyMuted;
135  }
136 
137  // the track could be locked, but we only show the 'locked portion' on the upper most track
138  // that is causing the lock
139  if (content == null)
140  return;
141 
142  var textRect = trackRect;
143  textRect.width = styles.fontClip.CalcSize(content).x + k_LockTextPadding;
144  textRect.x += (trackRect.width - textRect.width) / 2f;
145  textRect.height -= 4f;
146  textRect.y += 2f;
147 
148 
149  TimelineWindow.instance.OverlayDrawData.Add(TimelineWindow.OverlayData.CreateTextBoxOverlay(GUIClip.Unclip(textRect), content.text, styles.fontClip, Color.white, styles.customSkin.colorLockTextBG, styles.displayBackground));
