103  static void DoZoom(float zoomFactor, WindowState state, Vector2 refRange, float focalTime)
104  {
105  const float kMinRange = 0.05f; // matches zoomable area.
106 
107  var s = zoomFactor;
108  if (s <= 0) return;
109 
110  var t = Mathf.Max(focalTime, refRange.x);
111  var x = (refRange.x + t * (s - 1)) / s;
112  var y = (refRange.y + t * (s - 1)) / s;
113 
114  // don't set it if we reach the limit or panning happens
115  if (Math.Abs(x - y) > kMinRange)
116  {
117  // Zoomable area does not protect 100% against crazy values
118  state.SetTimeAreaShownRange(
119  Math.Max(x, -WindowConstants.timeAreaShownRangePadding),
120  Math.Min(y, WindowState.kMaxShownTime));
121  }
122  }

187  static float PixelToZoom(float x)
188  {
189  const float pixel2Zoom = 1 / 300.0f;
190  x *= pixel2Zoom;
191  if (x < -0.75)
192  {
193  // Rational function that behaves like 1+x on [-0.75,inf) and asymptotically reaches zero on (-inf,-0.75]
194  // The coefficients were obtained by the following constraints:
195  //1) f(-0.75) = 0.25
196  //2) f'(-0.75) = 1 C1 continuity
197  //3) f(-3) = 0.001 (asymptotically zero)
198  return 1 / (98.6667f + 268.444f * x + 189.63f * x * x);
199  }
200  return 1 + x;
201  }
