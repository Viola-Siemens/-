271  public void CalculateRowRects()
272  {
273  if (m_TreeView.isSearching)
274  return;
275 
276  const float startY = 6.0f;
277  IList<TreeViewItem> rows = m_TreeView.data.GetRows();
278  m_RowRects = new List<Rect>(rows.Count);
279  m_ExpandedRowRects = new List<Rect>(rows.Count);
280 
281  float curY = startY;
282  m_MaxWidthOfRows = 1f;
283 
284  // first pass compute the row rects
285  for (int i = 0; i < rows.Count; ++i)
286  {
287  var item = rows[i];
288 
289  if (i != 0)
290  curY += GetSpacing(item);
291 
292  Vector2 rowSize = GetSizeOfRow(item);
293  m_RowRects.Add(new Rect(0, curY, rowSize.x, rowSize.y));
294  m_ExpandedRowRects.Add(m_RowRects[i]);
295 
296  curY += rowSize.y;
297 
298  if (rowSize.x > m_MaxWidthOfRows)
299  m_MaxWidthOfRows = rowSize.x;
300 
301  // updated the expanded state
302  var groupGUI = item as TimelineGroupGUI;
303  if (groupGUI != null)
304  groupGUI.SetExpanded(m_TreeView.data.IsExpanded(item));
305  }
306 
307  float halfHeight = halfDropBetweenHeight;
308  const float kGroupPad = 1.0f;
309  const float kSkinPadding = 5.0f * 0.6f;
310  // work bottom up and compute visible regions for groups
311  for (int i = rows.Count - 1; i > 0; i--)
312  {
313  float height = 0;
314  TimelineTrackBaseGUI item = (TimelineTrackBaseGUI)rows[i];
315  if (item.isExpanded && item.children != null && item.children.Count > 0)
316  {
317  for (var j = 0; j < item.children.Count; j++)
318  {
319  var child = item.children[j];
320  int index = rows.IndexOf(child);
321  if (index > i)
322  height += m_ExpandedRowRects[index].height + kSkinPadding;
323  }
324 
325  height += kGroupPad;
326  }
327  m_ExpandedRowRects[i] = new Rect(m_RowRects[i].x, m_RowRects[i].y, m_RowRects[i].width, m_RowRects[i].height + height);
328 
329  var groupGUI = item as TimelineGroupGUI;
330  if (groupGUI != null)
331  {
332  var spacing = GetSpacing(item) + 1;
333  groupGUI.expandedRect = m_ExpandedRowRects[i];
334  groupGUI.rowRect = m_RowRects[i];
335  groupGUI.dropRect = new Rect(m_RowRects[i].x, m_RowRects[i].y - spacing, m_RowRects[i].width, m_RowRects[i].height + Mathf.Max(halfHeight, spacing));
336  }
