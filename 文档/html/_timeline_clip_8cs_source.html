714  static double GetExtrapolatedTime(double time, ClipExtrapolation mode, double duration)
715  {
716  if (duration == 0)
717  return 0;
718 
719  switch (mode)
720  {
721  case ClipExtrapolation.None:
722  break;
723 
724  case ClipExtrapolation.Loop:
725  if (time < 0)
726  time = duration - (-time % duration);
727  else if (time > duration)
728  time %= duration;
729  break;
730 
731  case ClipExtrapolation.Hold:
732  if (time < 0)
733  return 0;
734  if (time > duration)
735  return duration;
736  break;
737 
738  case ClipExtrapolation.PingPong:
739  if (time < 0)
740  {
741  time = duration * 2 - (-time % (duration * 2));
742  time = duration - Math.Abs(time - duration);
743  }
744  else
745  {
746  time = time % (duration * 2.0);
747  time = duration - Math.Abs(time - duration);
748  }
749  break;
750 
751  case ClipExtrapolation.Continue:
752  break;
753  }
754  return time;
755  }
