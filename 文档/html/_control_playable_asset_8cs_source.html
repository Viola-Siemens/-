284  internal void UpdateDurationAndLoopFlag(IList<PlayableDirector> directors, IList<ParticleSystem> particleSystems)
285  {
286  if (directors.Count == 0 && particleSystems.Count == 0)
287  return;
288 
289  const double invalidDuration = double.NegativeInfinity;
290 
291  var maxDuration = invalidDuration;
292  var supportsLoop = false;
293 
294  foreach (var director in directors)
295  {
296  if (director.playableAsset != null)
297  {
298  var assetDuration = director.playableAsset.duration;
299 
300  if (director.playableAsset is TimelineAsset && assetDuration > 0.0)
301  // Timeline assets report being one tick shorter than they actually are, unless they are empty
302  assetDuration = (double)((DiscreteTime)assetDuration).OneTickAfter();
303 
304  maxDuration = Math.Max(maxDuration, assetDuration);
305  supportsLoop = supportsLoop || director.extrapolationMode == DirectorWrapMode.Loop;
306  }
307  }
308 
309  foreach (var particleSystem in particleSystems)
310  {
311  maxDuration = Math.Max(maxDuration, particleSystem.main.duration);
312  supportsLoop = supportsLoop || particleSystem.main.loop;
313  }
314 
315  m_Duration = double.IsNegativeInfinity(maxDuration) ? PlayableBinding.DefaultDuration : maxDuration;
316  m_SupportLoop = supportsLoop;
317  }

349  {
350  if (director == null)

364  foreach (var ps in gameObject.GetComponentsInChildren<ParticleSystem>(true))
365  {
366  driver.AddFromName<ParticleSystem>(ps.gameObject, "randomSeed");
367  driver.AddFromName<ParticleSystem>(ps.gameObject, "autoRandomSeed");
368  }
369  }
370 
371  if (active)
372  {
373  driver.AddFromName(gameObject, "m_IsActive");
374  }
375 
376  if (updateITimeControl)
377  {
378  foreach (var script in GetControlableScripts(gameObject))
379  {
380  var propertyPreview = script as IPropertyPreview;
381  if (propertyPreview != null)
382  propertyPreview.GatherProperties(director, driver);
383  else
384  driver.AddFromComponent(script.gameObject, script);
385  }
386  }
387 
388  if (updateDirector)
389  {
390  foreach (var childDirector in GetComponent<PlayableDirector>(gameObject))
391  {
392  if (childDirector == null)
393  continue;
394 
395  var timeline = childDirector.playableAsset as TimelineAsset;
396  if (timeline == null)
397  continue;
398 
399  timeline.GatherProperties(childDirector, driver);
400  }
401  }
402  }
403  s_ProcessedDirectors.Remove(director);
