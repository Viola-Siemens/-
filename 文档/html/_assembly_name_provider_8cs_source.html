70  public bool IsInternalizedPackagePath(string path)

72  if (string.IsNullOrWhiteSpace(path))

74  return false;
75  }
76  var packageInfo = FindForAssetPath(path);
77  if (packageInfo == null)
78  {
79  return false;

81  var packageSource = packageInfo.source;
82  switch (packageSource)
83  {
84  case PackageSource.Embedded:
85  return !ProjectGenerationFlag.HasFlag(ProjectGenerationFlag.Embedded);
86  case PackageSource.Registry:
87  return !ProjectGenerationFlag.HasFlag(ProjectGenerationFlag.Registry);
88  case PackageSource.BuiltIn:

90  case PackageSource.Unknown:

92  case PackageSource.Local:

94  case PackageSource.Git:
