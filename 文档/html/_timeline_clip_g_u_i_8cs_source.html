584  void CalculateLoopRects(Rect trackRect, WindowState state)
585  {
586  if (!m_ClipViewDirty)
587  return;
588 
589  m_LoopRects.Clear();
590  if (clip.duration < WindowState.kTimeEpsilon)
591  return;
592 
593  var times = TimelineHelpers.GetLoopTimes(clip);
594  var loopDuration = TimelineHelpers.GetLoopDuration(clip);
595  m_MinLoopIndex = -1;
596 
597  // we have a hold, no need to compute all loops
598  if (!supportsLooping)
599  {
600  if (times.Length > 1)
601  {
602  var t = times[1];
603  float loopTime = (float)(clip.duration - t);
604  m_LoopRects.Add(ProjectRectOnTimeline(new Rect((float)(t + clip.start), 0, loopTime, 0), trackRect, state));
605  }
606  return;
607  }
608 
609  var range = state.timeAreaShownRange;
610  var visibleStartTime = range.x - clip.start;
611  var visibleEndTime = range.y - clip.start;
612 
613  for (int i = 1; i < times.Length; i++)
614  {
615  var t = times[i];
616 
617  // don't draw off screen loops
618  if (t > visibleEndTime)
619  break;
620 
621  float loopTime = Mathf.Min((float)(clip.duration - t), (float)loopDuration);
622  var loopEnd = t + loopTime;
623 
624  if (loopEnd < visibleStartTime)
625  continue;
626 
627  m_LoopRects.Add(ProjectRectOnTimeline(new Rect((float)(t + clip.start), 0, loopTime, 0), trackRect, state));
628 
629  if (m_MinLoopIndex == -1)
630  m_MinLoopIndex = i;
631  }
632  }

647  public IEnumerable<Edge> SnappableEdgesFor(IAttractable attractable, ManipulateEdges manipulateEdges)
648  {
649  var edges = new List<Edge>();
650 
651  bool canAddEdges = !parent.muted;
652 
653  if (canAddEdges)
654  {
655  // Hack: Trim Start in Ripple mode should not have any snap point added
656  if (EditMode.editType == EditMode.EditType.Ripple && manipulateEdges == ManipulateEdges.Left)
657  return edges;
658 
659  if (attractable != this)
660  {
661  if (EditMode.editType == EditMode.EditType.Ripple)
662  {
663  bool skip = false;
664 
665  // Hack: Since Trim End and Move in Ripple mode causes other snap point to move on the same track (which is not supported), disable snapping for this special cases...
666  // TODO Find a proper way to have different snap edges for each edit mode.
667  if (manipulateEdges == ManipulateEdges.Right)
668  {
669  var otherClipGUI = attractable as TimelineClipGUI;
670  skip = otherClipGUI != null && otherClipGUI.parent == parent;
671  }
672  else if (manipulateEdges == ManipulateEdges.Both)
673  {
674  var moveHandler = attractable as MoveItemHandler;
675  skip = moveHandler != null && moveHandler.movingItems.Any(clips => clips.targetTrack == clip.parentTrack && clip.start >= clips.start);
676  }
677 
678  if (skip)
679  return edges;
680  }
681 
682  AddEdge(edges, clip.start);
683  AddEdge(edges, clip.end);
684  }
685  else
686  {
687  if (manipulateEdges == ManipulateEdges.Right)
688  {
689  var d = TimelineHelpers.GetClipAssetEndTime(clip);
690 
691  if (d < double.MaxValue)
692  {
693  if (clip.SupportsLooping())
694  {
695  var l = TimelineHelpers.GetLoopDuration(clip);
696 
697  var shownTime = TimelineWindow.instance.state.timeAreaShownRange;
698  do
699  {
700  AddEdge(edges, d, false);
701  d += l;
702  }
703  while (d < shownTime.y);
704  }
705  else
706  {
707  AddEdge(edges, d, false);
708  }
709  }
710  }
711 
712  if (manipulateEdges == ManipulateEdges.Left)
713  {
714  var clipInfo = AnimationClipCurveCache.Instance.GetCurveInfo(clip.animationClip);
715  if (clipInfo != null && clipInfo.keyTimes.Any())
716  AddEdge(edges, clip.FromLocalTimeUnbound(clipInfo.keyTimes.Min()), false);
717  }
718  }
719  }
720  return edges;
721  }
