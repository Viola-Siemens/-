36  public static bool TrackHasAnimationCurves(TrackAsset track)
37  {
38  if (track.hasCurves)
39  return true;
40 
41  var animTrack = track as AnimationTrack;
42  if (animTrack != null && animTrack.infiniteClip != null)
43  return true;
44 
45  for (int i = 0; i < track.clips.Length; i++)
46  {
47  var curveClip = track.clips[i].curves;
48  var animationClip = track.clips[i].animationClip;
49 
50  // prune out clip with zero curves
51  if (curveClip != null && curveClip.empty)
52  curveClip = null;
53 
54  if (animationClip != null && animationClip.empty)
55  animationClip = null;
56 
57  // prune out clips coming from FBX
58  if (animationClip != null && ((animationClip.hideFlags & HideFlags.NotEditable) != 0))
59  animationClip = null;
60 
61  if (!track.clips[i].recordable)
62  animationClip = null;
63 
64  if ((curveClip != null) || (animationClip != null))
65  return true;
66  }
67 
68  return false;
