61  public static PlacementValidity GetPlacementValidity(ITimelineItem item, IEnumerable<ITimelineItem> otherItems)
62  {
63  if (item.duration <= 0.0)
64  return PlacementValidity.Valid; //items without any duration can always be placed
65 
66  var sortedItems = otherItems.Where(i => i.duration > 0.0).OrderBy(c => c.start);
67  var candidates = new List<ITimelineItem>();
68  foreach (var sortedItem in sortedItems)
69  {
70  if ((DiscreteTime)sortedItem.start >= (DiscreteTime)item.end)
71  {
72  // No need to process further
73  break;
74  }
75 
76  if ((DiscreteTime)sortedItem.end <= (DiscreteTime)item.start)
77  {
78  // Skip
79  continue;
80  }
81 
82  candidates.Add(sortedItem);
83  }
84 
85  var discreteStart = (DiscreteTime)item.start;
86  var discreteEnd = (DiscreteTime)item.end;
87 
88  // Note: Order of tests matters
89  for (int i = 0, n = candidates.Count; i < n; i++)
90  {
91  var candidate = candidates[i];
92 
93  var blendItem = item as IBlendable;
94  if (blendItem != null && blendItem.supportsBlending)
95  {
96  if (EditModeUtils.Contains(candidate.start, candidate.end, item))
97  return PlacementValidity.InvalidIsWithin;
98 
99  if (i < n - 1)
100  {
101  var nextCandidate = candidates[i + 1];
102 
103  var discreteNextCandidateStart = (DiscreteTime)nextCandidate.start;
104  var discreteCandidateEnd = (DiscreteTime)candidate.end;
105 
106  if (discreteCandidateEnd > discreteNextCandidateStart)
107  {
108  if (discreteStart >= discreteNextCandidateStart)
109  {
110  // Note: In case the placement is fully within a blend,
111  // InvalidStartsInBlend MUST have priority
112  return PlacementValidity.InvalidStartsInBlend;
113  }
114 
115  if (discreteEnd > discreteNextCandidateStart && discreteEnd <= discreteCandidateEnd)
116  return PlacementValidity.InvalidEndsInBlend;
117 
118  if (discreteStart < discreteNextCandidateStart && discreteEnd > discreteCandidateEnd)
119  return PlacementValidity.InvalidContainsBlend;
120  }
121  }
122 
123  if (EditModeUtils.Contains(item.start, item.end, candidate))
124  return PlacementValidity.InvalidContains;
125  }
126  else
127  {
128  if (EditModeUtils.Overlaps(item, candidate.start, candidate.end)
129  || EditModeUtils.Overlaps(candidate, item.start, item.end))
130  return PlacementValidity.InvalidOverlapWithNonBlendableClip;
131  }
132  }
