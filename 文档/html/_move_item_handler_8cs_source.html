38  public void Grab(IEnumerable<ITimelineItem> items, TrackAsset referenceTrack, Vector2 mousePosition)
39  {
40  if (items == null) return;
41 
42  items = items.ToArray(); // Cache enumeration result
43 
44  if (!items.Any()) return;
45 
46  m_GrabbedModalUndoGroup = Undo.GetCurrentGroup();
47 
48  var trackItems = items.GroupBy(c => c.parentTrack).ToArray();
49  var trackItemsCount = trackItems.Length;
50  var tracks = items.Select(c => c.parentTrack).Where(x => x != null).Distinct();
51 
52  movingItems = new MovingItems[trackItemsCount];
53 
54  allowTrackSwitch = trackItemsCount == 1 && !trackItems.SelectMany(x => x).Any(x => x is MarkerItem); // For now, track switch is only supported when all items are on the same track and there are no items
55  foreach (var sourceTrack in tracks)
56  {
57  // one push per track handles all the clips on the track
58  TimelineUndo.PushUndo(sourceTrack, "Move Items");
59 
60  // push all markers on the track because of ripple
61  foreach (var marker in sourceTrack.GetMarkers().OfType<ScriptableObject>())
62  TimelineUndo.PushUndo(marker, "Move Items");
63  }
64 
65  for (var i = 0; i < trackItemsCount; ++i)
66  {
67  var track = trackItems[i].Key;
68  var grabbedItems = new MovingItems(m_State, track, trackItems[i].ToArray(), referenceTrack, mousePosition, allowTrackSwitch);
69  movingItems[i] = grabbedItems;
70  }
71 
72  m_LeftMostMovingItems = null;
73  m_RightMostMovingItems = null;
74 
75  foreach (var grabbedTrackItems in movingItems)
76  {
77  if (m_LeftMostMovingItems == null || m_LeftMostMovingItems.start > grabbedTrackItems.start)
78  m_LeftMostMovingItems = grabbedTrackItems;
79 
80  if (m_RightMostMovingItems == null || m_RightMostMovingItems.end < grabbedTrackItems.end)
81  m_RightMostMovingItems = grabbedTrackItems;
82  }
83 
84  m_ItemGUIs = new HashSet<TimelineItemGUI>();
85  m_ItemsGroup = new ItemsGroup(items);
86 
87  foreach (var item in items)
88  m_ItemGUIs.Add(item.gui);
89 
90  targetTrack = referenceTrack;
91

93  m_Grabbing = true;
94  }
