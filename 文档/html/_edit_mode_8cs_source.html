274  public static void HandleModeClutch()
275  {
276  if (Event.current.type == EventType.KeyDown && EditorGUI.IsEditingTextField())
277  return;
278 
279  var prevType = editType;
280 
281  if (Event.current.type == EventType.KeyDown)
282  {
283  var clutchEditType = GetSubEditType(Event.current.keyCode);
284  if (clutchEditType != EditType.None)
285  {
286  s_OverrideEditType = clutchEditType;
287  Event.current.Use();
288  }
289  }
290  else if (Event.current.type == EventType.KeyUp)
291  {
292  var clutchEditType = GetSubEditType(Event.current.keyCode);
293  if (clutchEditType == s_OverrideEditType)
294  {
295  s_OverrideEditType = EditType.None;
296  Event.current.Use();
297  }
298  }
299 
300  if (prevType != editType)
301  {
302  if (s_CurrentTrimItem != null)
303  {
304  trimMode.OnBeforeTrim(s_CurrentTrimItem, s_CurrentTrimDirection);
305  }
306  else if (s_CurrentMoveItemHandler != null)
307  {
308  if (s_CurrentMoveItemHandler.movingItems == null)
309  {
310  s_CurrentMoveItemHandler = null;
311  return;
312  }
313 
314  foreach (var movingItems in s_CurrentMoveItemHandler.movingItems)
315  {
316  if (movingItems != null && movingItems.HasAnyDetachedParents())
317  {
318  foreach (var items in movingItems.items)
319  {
320  items.parentTrack = movingItems.originalTrack;
321  }
322  }
323  }
324 
325  var movingSelection = s_CurrentMoveItemHandler.movingItems;
326 
327  // Handle clutch key transition if needed
328  GetSubEditMode(prevType).moveItemMode.OnModeClutchExit(movingSelection);
329  moveMode.OnModeClutchEnter(movingSelection);
330 
331  moveMode.BeginMove(movingSelection);
332  moveMode.HandleTrackSwitch(movingSelection);
333 
334  UpdateMove();
335  s_CurrentMoveItemHandler.RefreshPreviewItems();
336 
337  TimelineWindow.instance.state.rebuildGraph = true; // TODO Rebuild only if parent changed
338  }
339 
340  TimelineWindow.instance.Repaint(); // TODO Refresh the toolbar without doing a full repaint?
341  }
342  }
