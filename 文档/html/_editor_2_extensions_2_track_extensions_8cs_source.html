86  internal static void UpdateClipIntersection(TimelineClip blendOutClip, TimelineClip blendInClip)
87  {
88  if (!blendOutClip.SupportsBlending() || !blendInClip.SupportsBlending())
89  return;
90 
91  if (blendInClip.end < blendOutClip.end)
92  return;
93 
94  double duration = Math.Max(0, blendOutClip.start + blendOutClip.duration - blendInClip.start);
95  duration = duration <= kMinOverlapTime ? 0 : duration;
96  blendOutClip.blendOutDuration = duration;

143  internal static TrackAsset Duplicate(this TrackAsset track, IExposedPropertyTable sourceTable, IExposedPropertyTable destTable,
144  TimelineAsset destinationTimeline = null)
145  {
146  if (track == null)
147  return null;
148 
149  // if the destination is us, clear to avoid bad parenting (case 919421)
150  if (destinationTimeline == track.timelineAsset)
151  destinationTimeline = null;
152 
153  var timelineParent = track.parent as TimelineAsset;
154  var trackParent = track.parent as TrackAsset;
155  if (timelineParent == null && trackParent == null)
156  {
157  Debug.LogWarning("Cannot duplicate track because it is not parented to known type");
158  return null;
159  }
160 
161  // Determine who the final parent is. If we are pasting into another track, it's always the timeline.
162  // Otherwise it's the original parent
163  PlayableAsset finalParent = destinationTimeline != null ? destinationTimeline : track.parent;
164 
165  // grab the list of tracks to generate a name from (923360) to get the list of names
166  // no need to do this part recursively
167  var finalTrackParent = finalParent as TrackAsset;
168  var finalTimelineAsset = finalParent as TimelineAsset;
169  var otherTracks = (finalTimelineAsset != null) ? finalTimelineAsset.trackObjects : finalTrackParent.subTracksObjects;
170 
171  // Important to create the new objects before pushing the original undo, or redo breaks the
172  // sequence
173  var newTrack = TimelineHelpers.Clone(finalParent, track, sourceTable, destTable, finalParent);
174  newTrack.name = TimelineCreateUtilities.GenerateUniqueActorName(otherTracks, newTrack.name);
175 
176  RecursiveSubtrackClone(track, newTrack, sourceTable, destTable, finalParent);
177  TimelineCreateUtilities.SaveAssetIntoObject(newTrack, finalParent);
178  TimelineUndo.RegisterCreatedObjectUndo(newTrack, "Duplicate");
179  TimelineUndo.PushUndo(finalParent, "Duplicate");
180 
181  if (destinationTimeline != null) // other timeline
182  destinationTimeline.AddTrackInternal(newTrack);
183  else if (timelineParent != null) // this timeline, no parent
184  ReparentTracks(new List<TrackAsset> { newTrack }, timelineParent, timelineParent.GetRootTracks().Last(), false);
185  else // this timeline, with parent
186  trackParent.AddChild(newTrack);
187 
188  // Call the custom editor. this check prevents the call when copying to the clipboard
189  if (destinationTimeline == null || destinationTimeline == TimelineEditor.inspectedAsset)
190  {
191  var customEditor = CustomTimelineEditorCache.GetTrackEditor(newTrack);
192  try
193  {
194  customEditor.OnCreate(newTrack, track);
195  }
196  catch (Exception e)
197  {
198  Debug.LogException(e);
199  }
200  }
201 
202  return newTrack;

205  // Reparents a list of tracks to a new parent
206  // the new parent cannot be null (has to be track asset or sequence)
207  // the insertAfter can be null (will not reorder)
208  internal static bool ReparentTracks(List<TrackAsset> tracksToMove, PlayableAsset targetParent,
209  TrackAsset insertMarker = null, bool insertBefore = false)
210  {
211  var targetParentTrack = targetParent as TrackAsset;
212  var targetSequenceTrack = targetParent as TimelineAsset;
213 
214  if (tracksToMove == null || tracksToMove.Count == 0 || (targetParentTrack == null && targetSequenceTrack == null))
215  return false;
216 
217  // invalid parent type on a track
218  if (targetParentTrack != null && tracksToMove.Any(x => !TimelineCreateUtilities.ValidateParentTrack(targetParentTrack, x.GetType())))
219  return false;
220 
221  // no valid tracks means this is simply a rearrangement
222  var validTracks = tracksToMove.Where(x => x.parent != targetParent).ToList();
223  if (insertMarker == null && !validTracks.Any())
224  return false;
225 
226  var parents = validTracks.Select(x => x.parent).Where(x => x != null).Distinct().ToList();
227 
228  // push the current state of the tracks that will change
229  foreach (var p in parents)
230  {
231  TimelineUndo.PushUndo(p, "Reparent");
232  }
233  foreach (var t in validTracks)
234  {
235  TimelineUndo.PushUndo(t, "Reparent");
236  }
237  TimelineUndo.PushUndo(targetParent, "Reparent");
238 
239  // need to reparent tracks first, before moving them.
240  foreach (var t in validTracks)
241  {
242  if (t.parent != targetParent)
243  {
244  TrackAsset toMoveParent = t.parent as TrackAsset;
245  TimelineAsset toMoveTimeline = t.parent as TimelineAsset;
246  if (toMoveTimeline != null)
247  {
248  toMoveTimeline.RemoveTrack(t);
249  }
250  else if (toMoveParent != null)
251  {
252  toMoveParent.RemoveSubTrack(t);
253  }
254 
255  if (targetParentTrack != null)
256  {
257  targetParentTrack.AddChild(t);
258  targetParentTrack.SetCollapsed(false);
259  }
260  else
261  {
262  targetSequenceTrack.AddTrackInternal(t);
263  }
264  }
265  }
266 
267 
268  if (insertMarker != null)
269  {
270  // re-ordering track. This is using internal APIs, so invalidation of the tracks must be done manually to avoid
271  // cache mismatches
272  var children = targetParentTrack != null ? targetParentTrack.subTracksObjects : targetSequenceTrack.trackObjects;
273  TimelineUtility.ReorderTracks(children, tracksToMove, insertMarker, insertBefore);
274  if (targetParentTrack != null)
275  targetParentTrack.Invalidate();
276  if (insertMarker.timelineAsset != null)
277  insertMarker.timelineAsset.Invalidate();
278  }
279 
280  return true;
281  }

414  // Find the gap at the given time
415  // return true if there is a gap, false if there is an intersection
416  // endGap will be Infinity if the gap has no end
417  internal static bool GetGapAtTime(this TrackAsset track, double time, out double startGap, out double endGap)
418  {
419  startGap = 0;
420  endGap = Double.PositiveInfinity;
421 
422  if (track == null || !track.GetClips().Any())
423  {
424  return false;
425  }
426 
427  var discreteTime = new DiscreteTime(time);
428 
429  track.SortClips();
430  var sortedByStartTime = track.clips;
431  for (int i = 0; i < sortedByStartTime.Length; i++)
432  {
433  var clip = sortedByStartTime[i];
434 
435  // intersection
436  if (discreteTime >= new DiscreteTime(clip.start) && discreteTime < new DiscreteTime(clip.end))
437  {
438  endGap = time;
439  startGap = time;
440  return false;
441  }
442 
443  if (clip.end < time)
444  {
445  startGap = clip.end;
446  }
447  if (clip.start > time)
448  {
449  endGap = clip.start;
450  break;
451  }
452  }
453 
454  if (endGap - startGap < TimelineClip.kMinDuration)
455  {
456  startGap = time;
457  endGap = time;
458  return false;
459  }
460 
461  return true;
462  }
