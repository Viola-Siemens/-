521  public void EnsurePlayHeadIsVisible()
522  {
523  double minDisplayedTime = PixelToTime(timeAreaRect.xMin);
524  double maxDisplayedTime = PixelToTime(timeAreaRect.xMax);
525 
526  double currentTime = editSequence.time;
527  if (currentTime >= minDisplayedTime && currentTime <= maxDisplayedTime)
528  return;
529 
530  float displayedTimeRange = (float)(maxDisplayedTime - minDisplayedTime);
531  float minimumTimeToDisplay = (float)currentTime - displayedTimeRange / 2.0f;
532  float maximumTimeToDisplay = (float)currentTime + displayedTimeRange / 2.0f;
533  SetTimeAreaShownRange(minimumTimeToDisplay, maximumTimeToDisplay);
534  }

536  public void SetPlayHeadToMiddle()
537  {
538  double minDisplayedTime = PixelToTime(timeAreaRect.xMin);
539  double maxDisplayedTime = PixelToTime(timeAreaRect.xMax);
540 
541  double currentTime = editSequence.time;
542  float displayedTimeRange = (float)(maxDisplayedTime - minDisplayedTime);
543 
544  if (currentTime >= minDisplayedTime && currentTime <= maxDisplayedTime)
545  {
546  if (currentTime < minDisplayedTime + displayedTimeRange / 2)
547  return;
548  }
549 
550  const float kCatchUpSpeed = 3f;
551  float realDelta = Mathf.Clamp(Time.realtimeSinceStartup - m_LastTime, 0f, 1f) * kCatchUpSpeed;
552  float scrollCatchupAmount = kCatchUpSpeed * realDelta * displayedTimeRange / 2;
553 
554  if (currentTime < minDisplayedTime)
555  {
556  SetTimeAreaShownRange((float)currentTime, (float)currentTime + displayedTimeRange);
557  }
558  else if (currentTime > maxDisplayedTime)
559  {
560  SetTimeAreaShownRange((float)currentTime - displayedTimeRange + scrollCatchupAmount, (float)currentTime + scrollCatchupAmount);
561  }
562  else if (currentTime > minDisplayedTime + displayedTimeRange / 2)
563  {
564  float targetMinDisplayedTime = Mathf.Min((float)minDisplayedTime + scrollCatchupAmount,
565  (float)(currentTime - displayedTimeRange / 2));
566  SetTimeAreaShownRange(targetMinDisplayedTime, targetMinDisplayedTime + displayedTimeRange);
567  }
568  }
