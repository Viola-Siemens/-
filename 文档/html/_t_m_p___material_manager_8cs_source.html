279  public static int GetStencilID(GameObject obj)
280  {
281  // Implementation is almost copied from Unity UI
282 
283  var count = 0;
284 
285  var transform = obj.transform;
286  var stopAfter = FindRootSortOverrideCanvas(transform);
287  if (transform == stopAfter)
288  return count;
289 
290  var t = transform.parent;
291  var components = TMP_ListPool<Mask>.Get();
292  while (t != null)
293  {
294  t.GetComponents<Mask>(components);
295  for (var i = 0; i < components.Count; ++i)
296  {
297  var mask = components[i];
298  if (mask != null && mask.MaskEnabled() && mask.graphic.IsActive())
299  {
300  ++count;
301  break;
302  }
303  }
304 
305  if (t == stopAfter)
306  break;
307 
308  t = t.parent;
309  }
310  TMP_ListPool<Mask>.Release(components);
311 
312  return Mathf.Min((1 << count) - 1, 255);
313  }

361  public static Material GetFallbackMaterial (Material sourceMaterial, Material targetMaterial)
362  {
363  int sourceID = sourceMaterial.GetInstanceID();
364  Texture tex = targetMaterial.GetTexture(ShaderUtilities.ID_MainTex);
365  int texID = tex.GetInstanceID();
366  long key = (long)sourceID << 32 | (long)(uint)texID;
367 
368  if (m_fallbackMaterials.TryGetValue(key, out FallbackMaterial fallback))
369  {
370  //Debug.Log("Material [" + fallback.fallbackMaterial.name + "] already exists.");
371  return fallback.fallbackMaterial;
372  }
373 
374  // Create new material from the source material and copy properties if using distance field shaders.
375  Material fallbackMaterial = null;
376  if (sourceMaterial.HasProperty(ShaderUtilities.ID_GradientScale) && targetMaterial.HasProperty(ShaderUtilities.ID_GradientScale))
377  {
378  fallbackMaterial = new Material(sourceMaterial);
379  fallbackMaterial.hideFlags = HideFlags.HideAndDontSave;
380 
381  #if UNITY_EDITOR
382  fallbackMaterial.name += " + " + tex.name;
383  //Debug.Log("Creating new fallback material for " + fallbackMaterial.name);
384  #endif
385 
386  fallbackMaterial.SetTexture(ShaderUtilities.ID_MainTex, tex);
387  // Retain material properties unique to target material.
388  fallbackMaterial.SetFloat(ShaderUtilities.ID_GradientScale, targetMaterial.GetFloat(ShaderUtilities.ID_GradientScale));
389  fallbackMaterial.SetFloat(ShaderUtilities.ID_TextureWidth, targetMaterial.GetFloat(ShaderUtilities.ID_TextureWidth));
390  fallbackMaterial.SetFloat(ShaderUtilities.ID_TextureHeight, targetMaterial.GetFloat(ShaderUtilities.ID_TextureHeight));
391  fallbackMaterial.SetFloat(ShaderUtilities.ID_WeightNormal, targetMaterial.GetFloat(ShaderUtilities.ID_WeightNormal));
392  fallbackMaterial.SetFloat(ShaderUtilities.ID_WeightBold, targetMaterial.GetFloat(ShaderUtilities.ID_WeightBold));
393  }
394  else
395  {
396  fallbackMaterial = new Material(targetMaterial);
397  }
398 
399  fallback = new FallbackMaterial();
400  fallback.baseID = sourceID;
401  fallback.baseMaterial = sourceMaterial;
402  fallback.fallbackID = key;
403  fallback.fallbackMaterial = fallbackMaterial;
404  fallback.count = 0;
405 
406  m_fallbackMaterials.Add(key, fallback);
407  m_fallbackMaterialLookup.Add(fallbackMaterial.GetInstanceID(), key);
408 
409  #if TMP_DEBUG_MODE
410  ListFallbackMaterials();
411  #endif
412 
413  return fallbackMaterial;
414  }

546  public static void CopyMaterialPresetProperties(Material source, Material destination)
547  {
548  if (!source.HasProperty(ShaderUtilities.ID_GradientScale) || !destination.HasProperty(ShaderUtilities.ID_GradientScale))
549  return;
550 
551  // Save unique material properties
552  Texture dst_texture = destination.GetTexture(ShaderUtilities.ID_MainTex);
553  float dst_gradientScale = destination.GetFloat(ShaderUtilities.ID_GradientScale);
554  float dst_texWidth = destination.GetFloat(ShaderUtilities.ID_TextureWidth);
555  float dst_texHeight = destination.GetFloat(ShaderUtilities.ID_TextureHeight);
556  float dst_weightNormal = destination.GetFloat(ShaderUtilities.ID_WeightNormal);
557  float dst_weightBold = destination.GetFloat(ShaderUtilities.ID_WeightBold);
558 
559  // Copy all material properties
560  destination.CopyPropertiesFromMaterial(source);
561 
562  // Copy shader keywords
563  destination.shaderKeywords = source.shaderKeywords;
564 
565  // Restore unique material properties
566  destination.SetTexture(ShaderUtilities.ID_MainTex, dst_texture);
567  destination.SetFloat(ShaderUtilities.ID_GradientScale, dst_gradientScale);
568  destination.SetFloat(ShaderUtilities.ID_TextureWidth, dst_texWidth);
569  destination.SetFloat(ShaderUtilities.ID_TextureHeight, dst_texHeight);
570  destination.SetFloat(ShaderUtilities.ID_WeightNormal, dst_weightNormal);
