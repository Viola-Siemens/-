374  void DoOffsetManipulator()
375  {
376  if (targets.Length > 1) //do not edit the track offset on a multiple selection
377  return;
378 
379  if (timelineWindow == null || timelineWindow.state == null || timelineWindow.state.editSequence.director == null)
380  return;
381 
382  AnimationTrack animationTrack = target as AnimationTrack;
383  if (animationTrack != null && (animationTrack.trackOffset == TrackOffset.ApplyTransformOffsets) && m_OffsetEditMode != TimelineAnimationUtilities.OffsetEditMode.None)
384  {
385  var boundObject = TimelineUtility.GetSceneGameObject(timelineWindow.state.editSequence.director, animationTrack);
386  var boundObjectTransform = boundObject != null ? boundObject.transform : null;
387 
388  var offsets = TimelineAnimationUtilities.GetTrackOffsets(animationTrack, boundObjectTransform);
389  EditorGUI.BeginChangeCheck();
390 
391  switch (m_OffsetEditMode)
392  {
393  case TimelineAnimationUtilities.OffsetEditMode.Translation:
394  offsets.position = Handles.PositionHandle(offsets.position, (Tools.pivotRotation == PivotRotation.Global)
395  ? Quaternion.identity
396  : offsets.rotation);
397  break;
398  case TimelineAnimationUtilities.OffsetEditMode.Rotation:
399  offsets.rotation = Handles.RotationHandle(offsets.rotation, offsets.position);
400  break;
401  }
402 
403  if (EditorGUI.EndChangeCheck())
404  {
405  TimelineUndo.PushUndo(animationTrack, "Inspector");
406  TimelineAnimationUtilities.UpdateTrackOffset(animationTrack, boundObjectTransform, offsets);
407  Evaluate();
408  Repaint();
409  }
410  }
411  }
