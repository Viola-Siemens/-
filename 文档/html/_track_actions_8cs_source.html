247  public static void SetLockState(TrackAsset[] tracks, bool shouldLock, WindowState state = null)
248  {
249  if (tracks.Length == 0)
250  return;
251 
252  foreach (var track in tracks)
253  {
254  if (TimelineUtility.IsLockedFromGroup(track))
255  continue;
256 
257  if (track as GroupTrack == null)
258  SetLockState(track.GetChildTracks().ToArray(), shouldLock, state);
259 
260  TimelineUndo.PushUndo(track, "Lock Tracks");
261  track.locked = shouldLock;
262  }
263 
264  if (state != null)
265  {
266  // find the tracks we've locked. unselect anything locked and remove recording.
267  foreach (var track in tracks)
268  {
269  if (TimelineUtility.IsLockedFromGroup(track) || !track.locked)
270  continue;
271 
272  var flattenedChildTracks = track.GetFlattenedChildTracks();
273  foreach (var i in track.clips)
274  SelectionManager.Remove(i);
275  state.UnarmForRecord(track);
276  foreach (var child in flattenedChildTracks)
277  {
278  SelectionManager.Remove(child);
279  state.UnarmForRecord(child);
280  foreach (var clip in child.GetClips())
281  SelectionManager.Remove(clip);
282  }
283  }
284 
285  // no need to rebuild, just repaint (including inspectors)
286  InspectorWindow.RepaintAllInspectors();
287  state.editorWindow.Repaint();
288  }
