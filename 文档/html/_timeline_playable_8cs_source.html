252  private void Evaluate(Playable playable, FrameData frameData)
253  {
254  if (m_IntervalTree == null)
255  return;
256 
257  double localTime = playable.GetTime();
258  m_ActiveBit = m_ActiveBit == 0 ? 1 : 0;
259 
260  m_CurrentListOfActiveClips.Clear();
261  m_IntervalTree.IntersectsWith(DiscreteTime.GetNearestTick(localTime), m_CurrentListOfActiveClips);
262 
263  foreach (var c in m_CurrentListOfActiveClips)
264  {
265  c.intervalBit = m_ActiveBit;
266  if (frameData.timeLooped)
267  c.Reset();
268  }
269 
270  // all previously active clips having a different intervalBit flag are not
271  // in the current intersection, therefore are considered becoming disabled at this frame
272  var timelineEnd = playable.GetDuration();
273  foreach (var c in m_ActiveClips)
274  {
275  if (c.intervalBit != m_ActiveBit)
276  {
277  var clipEnd = (double)DiscreteTime.FromTicks(c.intervalEnd);
278  var time = frameData.timeLooped ? Math.Min(clipEnd, timelineEnd) : Math.Min(localTime, clipEnd);
279  c.EvaluateAt(time, frameData);
280  c.enable = false;
281  }
282  }
283 
284  m_ActiveClips.Clear();
285  // case 998642 - don't use m_ActiveClips.AddRange, as in 4.6 .Net scripting it causes GC allocs
286  for (var a = 0; a < m_CurrentListOfActiveClips.Count; a++)
287  {
288  m_CurrentListOfActiveClips[a].EvaluateAt(localTime, frameData);
289  m_ActiveClips.Add(m_CurrentListOfActiveClips[a]);
290  }
291 
292  int count = m_EvaluateCallbacks.Count;
293  for (int i = 0; i < count; i++)
294  {
295  m_EvaluateCallbacks[i].Evaluate();
296  }
297  }
