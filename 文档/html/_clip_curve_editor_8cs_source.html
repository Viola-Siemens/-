112  internal void OnCurvesUpdated()
113  {
114  if (m_DataSource == null)
115  return;
116 
117  if (m_CurveEditor == null)
118  return;
119 
120  if (m_CurveEditor.animationCurves.Length == 0)
121  return;
122 
123  List<CurveWrapper> curvesToUpdate = m_CurveEditor.animationCurves.Where(c => c.changed).ToList();
124 
125  // nothing changed, return.
126  if (curvesToUpdate.Count == 0)
127  return;
128 
129  AnimationClip clip = m_DataSource.animationClip;
130 
131  // something changed, manage the undo properly.
132  Undo.RegisterCompleteObjectUndo(clip, "Edit Clip Curve");
133 
134  foreach (CurveWrapper c in curvesToUpdate)
135  {
136  AnimationUtility.SetEditorCurve(clip, c.binding, c.curve);
137  c.changed = false;
138  }
139 
140  m_DataSource.UpdateCurves(curvesToUpdate);
141  }

181  void UpdateCurveEditorIfNeeded(WindowState state)
182  {
183  if ((Event.current.type != EventType.Layout) || (m_DataSource == null) || (m_BindingHierarchy == null) || (m_DataSource.animationClip == null))
184  return;
185 
186  AnimationClipCurveInfo curveInfo = AnimationClipCurveCache.Instance.GetCurveInfo(m_DataSource.animationClip);
187  int version = curveInfo.version;
188  if (version != m_LastClipVersion)
189  {
190  // tree has changed
191  if (m_LastCurveCount != curveInfo.curves.Length)
192  {
193  m_BindingHierarchy.RefreshTree();
194  m_LastCurveCount = curveInfo.curves.Length;
195  }
196  else
197  {
198  // update just the curves
199  m_BindingHierarchy.RefreshCurves();
200  }
201  m_LastClipVersion = version;
202  }
203 
204  if (state.timeInFrames)
205  m_CurveEditor.state = new FrameFormatCurveEditorState();
206  else
207  m_CurveEditor.state = new UnformattedCurveEditorState();
