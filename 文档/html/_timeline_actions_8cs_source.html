360  static double CalculateDuplicateTime(IEnumerable<ItemsPerTrack> duplicatedItems, Func<ITimelineItem, ITimelineItem, double> gapBetweenItems)
361  {
362  //Find the end time of the rightmost item
363  var itemsOnTracks = duplicatedItems.SelectMany(i => i.targetTrack.GetItems()).ToList();
364  var time = itemsOnTracks.Max(i => i.end);
365 
366  //From all the duplicated items, select the leftmost items
367  var firstDuplicatedItems = duplicatedItems.Select(i => i.leftMostItem);
368  var leftMostDuplicatedItems = firstDuplicatedItems.OrderBy(i => i.start).GroupBy(i => i.start).FirstOrDefault();
369  if (leftMostDuplicatedItems == null) return 0.0;
370 
371  foreach (var leftMostItem in leftMostDuplicatedItems)
372  {
373  var siblings = leftMostItem.parentTrack.GetItems();
374  var rightMostSiblings = siblings.OrderByDescending(i => i.end).GroupBy(i => i.end).FirstOrDefault();
375  if (rightMostSiblings == null) continue;
376 
377  foreach (var sibling in rightMostSiblings)
378  time = Math.Max(time, sibling.end + gapBetweenItems(leftMostItem, sibling));
379  }
380 
381  return time;
382  }
383  }

638  {

640  var frameBounds = selectionOnly ? curveEditor.GetSelectionBounds() : curveEditor.GetClipBounds();

642  var clipGUI = curveEditorOwner as TimelineClipGUI;
