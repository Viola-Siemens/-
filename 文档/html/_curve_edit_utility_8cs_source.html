531  public static int EvaluateIndex(ObjectReferenceKeyframe[] curve, float time)
532  {
533  if (curve == null || curve.Length == 0)
534  throw new InvalidOperationException("Can not evaluate a PPtr curve with no entries");
535 
536  // clamp conditions
537  if (time <= curve[0].time)
538  return 0;
539  if (time >= curve.Last().time)
540  return curve.Length - 1;
541 
542  // binary search
543  var max = curve.Length - 1;
544  var min = 0;
545  while (max - min > 1)
546  {
547  var imid = (min + max) / 2;
548  if (Mathf.Approximately(curve[imid].time, time))
549  return imid;
550  if (curve[imid].time < time)
551  min = imid;
552  else if (curve[imid].time > time)
553  max = imid;
554  }
555  return min;
556  }
